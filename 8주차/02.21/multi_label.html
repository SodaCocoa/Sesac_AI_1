# 이미지 제너레이터 정의
# TensorFlow의 Keras API에서 이미지 데이터 전처리를 위한 ImageDataGenerator 클래스를 임포트합니다.
from tensorflow.keras.preprocessing.image import ImageDataGenerator

# 훈련 데이터에 사용할 이미지 데이터 제너레이터를 생성합니다.
# rescale=1./255는 이미지 픽셀 값을 [0, 1] 범위로 정규화하기 위한 스케일링 인자입니다.
# 이는 모델이 이미지 데이터를 더 잘 학습할 수 있게 도와줍니다.
train_datagen = ImageDataGenerator(rescale=1. / 255)

# 검증 데이터에 사용할 이미지 데이터 제너레이터도 동일하게 생성합니다.
# 훈련 데이터와 검증 데이터 모두 동일한 스케일링을 적용해야 합니다.
val_datagen = ImageDataGenerator(rescale=1. / 255)

# 에폭당 스텝 수를 계산하는 함수를 정의합니다.
# 이 함수는 전체 샘플 수와 배치 크기를 입력으로 받아, 모든 샘플이 한 번씩 처리되기 위해
# 한 에폭 동안 수행해야 할 스텝(배치 처리 횟수)의 수를 계산합니다.
def get_steps(num_samples, batch_size):
    # 전체 샘플 수를 배치 크기로 나누었을 때 나머지가 있으면, 즉 모든 샘플이 균등하게
    # 배치에 포함되지 않으면, 한 번 더 배치 처리를 수행해야 합니다.
    if (num_samples % batch_size) > 0:
        return (num_samples // batch_size) + 1
    # 전체 샘플 수가 배치 크기로 나누어 떨어지면, 그 몫이 바로 필요한 스텝 수입니다.
    else:
        return num_samples // batch_size

# TensorFlow의 Keras API를 사용하여 신경망 모델을 구축하고 컴파일하는 과정입니다.

# 필요한 모듈을 임포트합니다.
# Sequential 모델은 층을 순서대로 쌓아 올릴 수 있게 해주는 클래스입니다.
# Dense는 완전 연결 층을, Flatten은 입력을 일렬로 펼치는 층을 의미합니다.
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, Flatten

# Sequential 모델의 인스턴스를 생성합니다.
model = Sequential()

# 첫 번째 층으로 Flatten을 추가합니다. 이 층은 모델의 입력을 일렬로 펼치는 역할을 합니다.
# input_shape는 모델에 입력되는 데이터의 형태를 정의합니다. 여기서는 112x112 크기의 RGB 이미지가 입력됩니다.
model.add(Flatten(input_shape=(112, 112, 3)))  # (112, 112, 3)의 형태를 (112 * 112 * 3)으로 펼칩니다.

# 완전 연결 층(Dense)을 추가합니다. 이 층은 128개의 뉴런을 가지며, 활성화 함수로 'relu'를 사용합니다.
model.add(Dense(128, activation='relu'))

# 또 다른 Dense 층을 추가합니다. 이 층은 64개의 뉴런을 가지며, 마찬가지로 'relu' 활성화 함수를 사용합니다.
model.add(Dense(64, activation='relu'))

# 마지막 Dense 층은 모델의 출력 층으로, 11개의 뉴런을 가집니다. 멀티 레이블 분류 문제이므로
# 각 뉴런은 하나의 레이블에 대한 예측을 나타내며, 'sigmoid' 활성화 함수를 사용합니다.
model.add(Dense(11, activation='sigmoid'))

# 모델을 컴파일합니다. 이 과정에서 손실 함수, 최적화 알고리즘, 평가 지표를 지정합니다.
# 'binary_crossentropy' 손실 함수는 멀티 레이블 분류에 적합합니다. 'adam' 최적화 알고리즘을 사용합니다.
# 평가 지표로는 정확도('acc')를 사용합니다.
model.compile(optimizer='adam', 
              loss='binary_crossentropy', 
              metrics=['acc'])

# 멀티 레이블 분류를 위해 예측할 클래스 목록을 정의합니다.
# 이 경우, 다양한 색상과 옷 종류에 해당하는 레이블입니다.
class_col = ['black', 'blue', 'brown', 'green', 'red', 'white', 'dress', 'shirt', 'pants', 'shorts', 'shoes']

# 훈련 데이터 생성기를 설정합니다.
# directory=None: 이미지 파일 경로가 DataFrame의 'image' 열에 이미 절대 경로로 제공되므로, directory 매개변수는 None으로 설정됩니다.
# x_col='image': 이미지 파일의 경로가 저장된 DataFrame 열의 이름입니다.
# y_col=class_col: 각 이미지에 대한 레이블이 저장된 열의 이름 목록입니다.
# target_size=(112, 112): 모델에 입력되기 전 이미지의 크기를 조정합니다.
# color_mode='rgb': 이미지를 RGB 모드로 로드합니다.
# class_mode='raw': 'raw' 모드를 사용하여 레이블 데이터가 그대로 반환되도록 합니다.
# batch_size=64: 한 번에 네트워크에 전달되는 이미지 수입니다.
# seed=42: 데이터 셔플링 시 사용되는 난수 시드입니다.
train_generator = train_datagen.flow_from_dataframe(
    dataframe=train_df,
    directory=None,
    x_col='image',
    y_col=class_col,
    target_size=(112, 112),
    color_mode='rgb',
    class_mode='raw',
    batch_size=64,
    seed=42
)

# 검증 데이터 생성기를 설정합니다.
# 검증 데이터에 대해서도 동일한 설정을 사용하지만, shuffle=True를 설정하여 데이터가 에폭마다 무작위로 섞이도록 합니다.
# 이는 모델이 검증 데이터에 대해 일반화되도록 돕습니다.
val_generator = val_datagen.flow_from_dataframe(
    dataframe=val_df,
    directory=None,
    x_col='image',
    y_col=class_col,
    target_size=(112, 112),
    color_mode='rgb',
    class_mode='raw',
    batch_size=64,
    shuffle=True
)

# 모델을 학습시작

# train_generator를 사용하여 모델을 학습합니다. train_generator는 학습 데이터를 배치 단위로 생성하는 제너레이터입니다.
# 한 에폭당 사용되는 단계 수는 len(train_df)를 배치 크기로 나눈 결과를 get_steps() 함수로 계산합니다.
# 여기서 64는 배치 크기를 나타냅니다.
history = model.fit(
    train_generator,
    steps_per_epoch=get_steps(len(train_df), 64),
    
    # val_generator를 사용하여 모델의 성능을 검증합니다. val_generator는 검증 데이터를 배치 단위로 생성하는 제너레이터입니다.
    # 한 에폭당 사용되는 검증 단계 수는 len(val_df)를 배치 크기로 나눈 결과를 get_steps() 함수로 계산합니다.
    # 여기서 64는 배치 크기를 나타냅니다.
    validation_data=val_generator,
    validation_steps=get_steps(len(val_df), 64),
    
    # 총 10 에폭 동안 학습을 수행합니다.
    epochs=10
)
